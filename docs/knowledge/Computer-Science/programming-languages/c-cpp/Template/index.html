class UniquePtr {
private:
    T* ptr;
    
public:
    // æ„é€ å‡½æ•°
    explicit UniquePtr(T* p = nullptr) : ptr(p) {}
    
    // ç§»åŠ¨æ„é€ å‡½æ•°

        other.ptr = nullptr;
    }
    
    // ç§»åŠ¨èµ‹å€¼æ“ä½œç¬¦
    UniquePtr& operator=(UniquePtr&& other) noexcept {
        if (this != &other) {

            ptr = other.ptr;
            other.ptr = nullptr;
        }
        return *this;
    }
    
    // ç¦ç”¨æ‹·è´

    UniquePtr& operator=(const UniquePtr&) = delete;
    
    // ææ„å‡½æ•°

        delete ptr;
    }
    
    // è®¿é—®æ“ä½œç¬¦
    T& operator*() const { return *ptr; }
    T* operator->() const { return ptr; }
    
    // è·å–åŸå§‹æŒ‡é’ˆ
    T* get() const { return ptr; }
    
    // é‡Šæ”¾æ‰€æœ‰æƒ
    T* release() {
        T* temp = ptr;
        ptr = nullptr;
        return temp;
    }
    
    // é‡ç½®æŒ‡é’ˆ
    void reset(T* p = nullptr) {
        delete ptr;
        ptr = p;
    }
    
    // å¸ƒå°”è½¬æ¢
    explicit operator bool() const {
        return ptr != nullptr;
    }
};
